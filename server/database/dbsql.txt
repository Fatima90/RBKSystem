CREATE TABLE IF NOT EXISTS `rbkadmission`.`agreementqustions` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `arabicQ` VARCHAR(200) NULL DEFAULT NULL,
  `englishQ` VARCHAR(200) NULL DEFAULT NULL,
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8



CREATE TABLE IF NOT EXISTS `rbkadmission`.`agreementanswer` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `userinfo_id` INT(11) NOT NULL,
  `agreementqustions_id` INT(11) NOT NULL,
  `answer` VARCHAR(200) NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  INDEX `userinfo_id` (`userinfo_id` ASC),
  INDEX `agreementqustions_id` (`agreementqustions_id` ASC),
  CONSTRAINT `agreementanswer_ibfk_1`
    FOREIGN KEY (`userinfo_id`)
    REFERENCES `rbkadmission`.`candidateinfo` (`id`),
  CONSTRAINT `agreementanswer_ibfk_2`
    FOREIGN KEY (`agreementqustions_id`)
    REFERENCES `rbkadmission`.`agreementqustions` (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8





CREATE TABLE IF NOT EXISTS `rbkadmission`.`candidateinfo` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(200) NULL DEFAULT NULL,
  `email` VARCHAR(200) NULL DEFAULT NULL,
  `password` VARCHAR(200) NULL DEFAULT NULL,
  `phone` VARCHAR(200) NULL DEFAULT NULL,
  `gender` VARCHAR(200) NULL DEFAULT NULL,
  `dateOfBirth` DATE NULL DEFAULT NULL,
  `nattionalaty` VARCHAR(200) NULL DEFAULT NULL,
  `residence` VARCHAR(200) NULL DEFAULT NULL,
  `destance` VARCHAR(200) NULL DEFAULT NULL,
  `camp_Location` VARCHAR(200) NULL DEFAULT NULL,
  `videoLink` VARCHAR(200) NULL DEFAULT NULL,
  `states` VARCHAR(200) NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `email` (`email` ASC))
ENGINE = InnoDB
AUTO_INCREMENT = 4
DEFAULT CHARACTER SET = utf8




CREATE TABLE IF NOT EXISTS `rbkadmission`.`usertest` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `userinfo_id` INT(11) NOT NULL,
  `personality` VARCHAR(200) NULL DEFAULT NULL,
  `mindSet` VARCHAR(200) NULL DEFAULT NULL,
  `analitical` VARCHAR(200) NULL DEFAULT NULL,
  `codeExp` TINYINT(1) NULL DEFAULT NULL,
  `codeSyntax` VARCHAR(200) NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  INDEX `userinfo_id` (`userinfo_id` ASC),
  CONSTRAINT `usertest_ibfk_1`
    FOREIGN KEY (`userinfo_id`)
    REFERENCES `rbkadmission`.`candidateinfo` (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
